    print(f"[Frida] {message}&&{data}")

try:

    session = device.attach("TEST")

    print("[+] 附加进程成功")

except frida.ProcessNotFoundError:

    print("[-] 进程未找到，请确保应用已启动")

    js_code=load_js_file()

    print(f"读取失败:{e}")

script.on('message',on_message)

script.load()
